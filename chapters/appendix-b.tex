\chapter{LaTeX Feature Reference}
\label{app:latex}

\emph{This appendix is a template cheat-sheet --- delete it before submission.}

\section{Citations}
Cite one source~\cite{smith2024example} or
several~\cite{smith2024example,tanaka2023example,lee2025example}.
Entries live in \texttt{references.bib}; the style is IEEE numeric.

\section{Figures}
Reference floats by label: \cref{fig:cheat-logo}.

\begin{figure}[ht]
  \centering
  \includegraphics[width=0.5\textwidth]{ist_logo.png}
  \caption{Caption goes \emph{below} a figure. Put image files in
           \texttt{figures/} and always add a \texttt{\textbackslash label}.}
  \label{fig:cheat-logo}
\end{figure}

\section{Subfigures (side-by-side)}
Place two images in one figure, each with its own sub-caption. Reference the whole
figure with \cref{fig:cheat-subs}, or a single panel with \cref{fig:cheat-sub-b}.

\begin{figure}[ht]
  \centering
  \begin{subfigure}[b]{0.42\textwidth}
    \centering
    \includegraphics[height=4cm]{PrincessTwilightSparkle.png}
    \caption{One pony.}
    \label{fig:cheat-sub-a}
  \end{subfigure}\hfill
  \begin{subfigure}[b]{0.42\textwidth}
    \centering
    \includegraphics[height=4cm]{Fluttershy.png}
    \caption{Another pony.}
    \label{fig:cheat-sub-b}
  \end{subfigure}
  \caption{Two images side by side, each with its own sub-caption.}
  \label{fig:cheat-subs}
\end{figure}
